#!/bin/bash
############################################################
# RNA-seq Processing Pipeline
# SRA Download → FASTQ → QC → Alignment → BAM → IGV
############################################################
#############################
# 1. INSTALL REQUIRED TOOLS
#############################
# Ubuntu/Linux installation

sudo apt update

sudo apt install -y \
sra-toolkit \
pigz \
fastqc \
multiqc \
samtools

# Install manually:
# Salmon  : https://combine-lab.github.io/salmon/
# STAR    : https://github.com/alexdobin/STAR
# IGV     : https://igv.org/app/

#############################
# 2. VERIFY INSTALLATION
#############################

prefetch --version
fasterq-dump --version
pigz --version
fastqc --version
multiqc --version
salmon --version
STAR --version
samtools --version

#############################
# 3. CREATE DIRECTORIES
#############################

mkdir -p fastq
mkdir -p fastqc_results
mkdir -p multiqc_results
mkdir -p star_index
mkdir -p star_alignment

#############################
# 4. DEFINE SRR IDS
#############################

SRR_IDS=(
  SRR23955797
  SRR23955798
  SRR23955799
)

#############################
# 5. DOWNLOAD SRA FILES
#############################

echo "======================================"
echo "Downloading SRA files"
echo "======================================"

for SRR in "${SRR_IDS[@]}"
do
    prefetch $SRR
done

#############################
# 6. CONVERT TO FASTQ
#############################

echo "======================================"
echo "Converting SRA to FASTQ"
echo "======================================"

for SRR in "${SRR_IDS[@]}"
do
    fasterq-dump $SRR \
    --split-files \
    --threads 4 \
    -O fastq
done

#############################
# 7. COMPRESS FASTQ FILES
#############################

echo "======================================"
echo "Compressing FASTQ files"
echo "======================================"

pigz -p 4 fastq/*.fastq

#############################
# 8. RUN FASTQC
#############################

echo "======================================"
echo "Running FastQC"
echo "======================================"

fastqc fastq/SRR23955797_1.fastq.gz -o fastqc_results/

fastqc fastq/SRR23955797_2.fastq.gz -o fastqc_results/

fastqc fastq/SRR23955798_1.fastq.gz -o fastqc_results/

fastqc fastq/SRR23955798_2.fastq.gz -o fastqc_results/

#############################
# 9. GENERATE MULTIQC REPORT
#############################

echo "======================================"
echo "Generating MultiQC Report"
echo "======================================"

multiqc fastqc_results/*.zip \
-o multiqc_results/

#############################
# 10. BUILD SALMON INDEX
#############################

# Download transcriptome.fa beforehand

echo "======================================"
echo "Building Salmon Index"
echo "======================================"

salmon index \
-t transcriptome.fa \
-i salmon_index \
-k 31

#############################
# 11. SALMON QUANTIFICATION
#############################

echo "======================================"
echo "Running Salmon Quantification"
echo "======================================"

for SRR in "${SRR_IDS[@]}"
do

salmon quant \
-i salmon_index \
-l A \
-1 fastq/${SRR}_1.fastq.gz \
-2 fastq/${SRR}_2.fastq.gz \
-p 12 \
--validateMappings \
-o ${SRR}_salmon_out

done

#############################
# 12. BUILD STAR GENOME INDEX
#############################

# Download beforehand:
# GRCh38.primary_assembly.genome.fa
# gencode.v49.annotation.gtf

echo "======================================"
echo "Building STAR Genome Index"
echo "======================================"

STAR \
--runThreadN 12 \
--runMode genomeGenerate \
--genomeDir star_index \
--genomeFastaFiles GRCh38.primary_assembly.genome.fa \
--sjdbGTFfile gencode.v49.annotation.gtf \
--sjdbOverhang 149

#############################
# 13. STAR ALIGNMENT
#############################

echo "======================================"
echo "Running STAR Alignment"
echo "======================================"

for SRR in "${SRR_IDS[@]}"
do

STAR \
--runThreadN 12 \
--genomeDir star_index \
--readFilesIn \
fastq/${SRR}_1.fastq.gz \
fastq/${SRR}_2.fastq.gz \
--readFilesCommand zcat \
--outSAMtype BAM SortedByCoordinate \
--outFileNamePrefix star_alignment/${SRR}_

done

#############################
# 14. CREATE BAM INDEX (.bai)
#############################

echo "======================================"
echo "Generating BAM Index Files"
echo "======================================"

for SRR in "${SRR_IDS[@]}"
do

samtools index \
star_alignment/${SRR}_Aligned.sortedByCoord.out.bam

done

#############################
# 15. BAM ALIGNMENT STATISTICS
#############################

echo "======================================"
echo "Running Samtools Statistics"
echo "======================================"

for SRR in "${SRR_IDS[@]}"
do

samtools flagstat \
star_alignment/${SRR}_Aligned.sortedByCoord.out.bam \
> star_alignment/${SRR}_flagstat.txt

done

#############################
# 16. IGV VISUALIZATION
#############################

# Open IGV
# Load Human hg38 genome
# Load:
# *.bam
# *.bam.bai

# Example files:
# SRR23955797_Aligned.sortedByCoord.out.bam
# SRR23955797_Aligned.sortedByCoord.out.bam.bai

# Search genes:
# MX1
# PLAU
# IFIT1
# STAT1
# TLR7


# Install:
# sudo apt install subread

echo "======================================"
echo "Generating Raw Gene Counts."
echo "======================================"

featureCounts \
-T 8 \
-p \
-a gencode.v49.annotation.gtf \
-o raw_gene_counts.txt \
star_alignment/*Aligned.sortedByCoord.out.bam




############################################################
# NOTE:
# If RNA-seq data is generated directly in the laboratory, the pipeline usually starts from FASTQ files instead of SRA download.
# Recommended workflow: FASTQ → FastQC/MultiQC → Trimmomatic (optional) → QC on trimmed FASTQ → Salmon/STAR Alignment
# Trimmomatic can be used to remove:
# - adapter contamination
# - poor-quality reads
# - low-quality bases
# After trimming, QC should again be performed on the
# cleaned FASTQ files before downstream analysis.
############################################################

echo "======================================"
echo "RNA-seq- BASH Pipeline Completed Successfully."
echo "======================================"
